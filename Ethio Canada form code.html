    // Bootstrap validation
    (function () {
      const form = document.getElementById('travellerForm');
      form.addEventListener('submit', function (e) {
        // Capture signatures
        document.getElementById('sigApplicantData').value = getCanvasData('sigApplicant');
        document.getElementById('sigGuardianData').value = getCanvasData('sigGuardian');

        if (!form.checkValidity() || !document.getElementById('sigApplicantData').value || !document.getElementById('sigGuardianData').value) {
          e.preventDefault();
          e.stopPropagation();
          form.classList.add('was-validated');
          // Nudge empty signature pads
          ['sigApplicant','sigGuardian'].forEach(id=>{
            const el = document.getElementById(id);
            if (isCanvasBlank(el)) { el.classList.add('border','border-danger'); }
          });
          return;
        }

        e.preventDefault();
        handleSubmit(new FormData(form));
      });
    })();

    // Signature pad
    function setupSigPad(canvasId) {
      const canvas = document.getElementById(canvasId);
      const ctx = canvas.getContext('2d');
      resizeCanvas(canvas);
      let drawing = false, last = null;

      const start = (x,y)=>{ drawing=true; last={x,y}; };
      const move  = (x,y)=>{
        if(!drawing) return;
        ctx.lineWidth = 2.2;
        ctx.lineCap = 'round';
        ctx.strokeStyle = '#cfe1ff';
        ctx.beginPath();
        ctx.moveTo(last.x,last.y);
        ctx.lineTo(x,y);
        ctx.stroke();
        last = {x,y};
      };
      const end = ()=>{ drawing=false; last=null; };

      canvas.addEventListener('mousedown', e=> start(e.offsetX, e.offsetY));
      canvas.addEventListener('mousemove', e=> move(e.offsetX, e.offsetY));
      canvas.addEventListener('mouseup', end);
      canvas.addEventListener('mouseleave', end);

      // Touch support
      canvas.addEventListener('touchstart', e=>{
        const r = canvas.getBoundingClientRect();
        const t = e.touches[0];
        start(t.clientX - r.left, t.clientY - r.top);
        e.preventDefault();
      }, {passive:false});
      canvas.addEventListener('touchmove', e=>{
        const r = canvas.getBoundingClientRect();
        const t = e.touches[0];
        move(t.clientX - r.left, t.clientY - r.top);
        e.preventDefault();
      }, {passive:false});
      canvas.addEventListener('touchend', end);

      // Clear button
      document.querySelectorAll(`[data-clear="${canvasId}"]`).forEach(btn=>{
        btn.addEventListener('click', ()=>{
          ctx.clearRect(0,0,canvas.width,canvas.height);
          canvas.classList.remove('border','border-danger');
          const hidden = document.getElementById(canvasId + 'Data');
          if (hidden) hidden.value = '';
        });
      });

      // Resize on window change
      window.addEventListener('resize', ()=> resizeCanvas(canvas));
    }

    function resizeCanvas(canvas) {
      const ratio = window.devicePixelRatio || 1;
      const style = getComputedStyle(canvas);
      const width = parseFloat(style.width);
      const height = parseFloat(style.height);
      canvas.width = Math.floor(width * ratio);
      canvas.height = Math.floor(height * ratio);
      const ctx = canvas.getContext('2d');
      ctx.scale(ratio, ratio);
    }

    function isCanvasBlank(canvas) {
      const ctx = canvas.getContext('2d');
      const pixelBuffer = new Uint32Array(
        ctx.getImageData(0, 0, canvas.width, canvas.height).data.buffer
      );
      return !pixelBuffer.some(color => color !== 0);
    }

    function getCanvasData(id) {
      const c = document.getElementById(id);
      if (isCanvasBlank(c)) return '';
      return c.toDataURL('image/png');
    }

    function handleSubmit(fd) {
      // Preview payload in console
      const obj = {};
      fd.forEach((v, k) => {
        if (v instanceof File) obj[k] = {name: v.name, size: v.size, type: v.type};
        else obj[k] = v;
      });
      console.log('Submitting payload:', obj);

      alert('መረጃዎ ተቀብሏል (በመጀመሪያ ለናገር ብቻ ይታያል). እባክዎ ከ console ውስጥ ይመልከቱ።');
    }

    // Init signature pads
